		<script>
			(function () {
				"use strict";

				var log = document.getElementById("log");
				var status = document.getElementById("status");
				var clearBtn = document.getElementById("clear");
				var MAX_VISIBLE = 500;
				var ws = null;
				var lastId = 0;
				var reconnectDelay = 500;
				var MAX_RECONNECT_DELAY = 5000;

				function kindClass(kind) {
					if (kind === "lsp_diagnostic") {
						return "kind-warning";
					}
					if (kind === "index_file_failed" || kind === "server_stopped") {
						return "kind-error";
					}
					if (kind === "index_done" || kind === "server_started") {
						return "kind-success";
					}
					if (kind === "ws_client_connected" || kind === "ws_client_disconnected") {
						return "kind-debug";
					}
					return "kind-info";
				}

				function formatTs(ts) {
					var d = new Date(ts);
					var hh = String(d.getHours()).padStart(2, "0");
					var mm = String(d.getMinutes()).padStart(2, "0");
					var ss = String(d.getSeconds()).padStart(2, "0");
					var ms = String(d.getMilliseconds()).padStart(3, "0");
					return hh + ":" + mm + ":" + ss + "." + ms;
				}

				function summarize(ev) {
					var parts = [];
					parts.push("[" + ev.kind + "]");
					switch (ev.kind) {
						case "lsp_progress":
							parts.push(ev.stage, ev.current + "/" + ev.total, ev.file);
							break;
						case "lsp_diagnostic":
							parts.push(ev.severity, ev.file + ":" + ev.line, ev.message);
							break;
						case "sqlite_stats":
							parts.push("pous=" + ev.pous, "vars=" + ev.variables, "types=" + ev.types, "rels=" + ev.relationships, "files=" + ev.files);
							break;
						case "index_started":
							parts.push(ev.workspace, "(" + ev.totalFiles + " files)");
							break;
						case "index_file_started":
							parts.push("[" + ev.index + "/" + ev.total + "]", "START", ev.file);
							break;
						case "index_file_done":
							parts.push("[" + ev.index + "/" + ev.total + "]", "DONE", ev.file, ev.entities + "e/" + ev.edges + "r", ev.durationMs + "ms");
							break;
						case "index_file_failed":
							parts.push("[" + ev.index + "/" + ev.total + "]", "FAIL", ev.file, ev.error);
							break;
						case "index_done":
							parts.push(ev.workspace, "indexed=" + ev.indexedFiles, "skipped=" + ev.skippedFiles, "entities=" + ev.totalEntities, "edges=" + ev.totalEdges, ev.totalTimeMs + "ms");
							break;
						case "ws_client_connected":
						case "ws_client_disconnected":
							parts.push(ev.remote);
							break;
						case "server_started":
							parts.push("pid=" + ev.pid, "bun=" + ev.bunVersion, "uiPort=" + ev.uiPort);
							break;
						case "server_stopped":
							parts.push("reason=" + ev.reason);
							break;
						case "bus_overflow":
							parts.push("dropped " + ev.dropped + "/" + ev.capacity);
							break;
					}
					return parts.join(" ");
				}

				function append(ev) {
					var row = document.createElement("div");
					row.className = "log";
					var id = document.createElement("span");
					id.className = "id";
					id.textContent = "#" + ev.id;
					var ts = document.createElement("span");
					ts.className = "ts";
					ts.textContent = formatTs(ev.ts);
					var kind = document.createElement("span");
					kind.className = "kind " + kindClass(ev.kind);
					kind.textContent = summarize(ev);
					row.appendChild(id);
					row.appendChild(ts);
					row.appendChild(kind);
					log.appendChild(row);
					while (log.childElementCount > MAX_VISIBLE) {
						log.removeChild(log.firstChild);
					}
					if (ev.id > lastId) lastId = ev.id;
				}

				function setStatus(text, cls) {
					status.textContent = text;
					status.className = cls;
				}

				function connect() {
					setStatus("connecting", "connecting");
					var proto = window.location.protocol === "https:" ? "wss:" : "ws:";
					var url = proto + "//" + window.location.host + "/ws";
					ws = new WebSocket(url);

					ws.onopen = function () {
						setStatus("connected", "connected");
						reconnectDelay = 500;
						// request any events we missed (since lastId)
						ws.send(JSON.stringify({ type: "replay", sinceId: lastId }));
					};

					ws.onmessage = function (msg) {
						try {
							var payload = JSON.parse(msg.data);
							if (payload.type === "events" && Array.isArray(payload.events)) {
								for (var i = 0; i < payload.events.length; i++) {
									append(payload.events[i]);
								}
							} else if (payload.type === "event") {
								append(payload.event);
							}
						} catch (e) {
							console.error("parse error", e);
						}
					};

					ws.onclose = function () {
						setStatus("disconnected", "disconnected");
						ws = null;
						setTimeout(connect, reconnectDelay);
						reconnectDelay = Math.min(MAX_RECONNECT_DELAY, reconnectDelay * 2);
					};

					ws.onerror = function () {
						// onclose will fire next; nothing to do here.
					};
				}

				clearBtn.addEventListener("click", function () {
					while (log.firstChild) log.removeChild(log.firstChild);
				});

				connect();
			})();
		</script>
